from json import loads
from types import SimpleNamespace

def get_latest_github_release(owner: str, project: str):
    response = request.urlopen('https://api.github.com/repos/{owner}/{project}/releases/latest'.format(owner=owner,
project=project))
    return SimpleNamespace(**loads(response.read()))
